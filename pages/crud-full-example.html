Full CRUD Project: ระบบจัดการงาน (Task Manager)

ในบทสุดท้ายนี้ เราจะนำทุกอย่างที่เรียนมาประกอบเป็นระบบ CRUD ที่ใช้งานได้จริง

Step 1: เตรียม Model & Table

php artisan make:model Task -m

ในไฟล์ Migration:

public function up() {
    Schema::create('tasks', function (Blueprint $table) {
        $table->id();
        $table->string('title');
        $table->boolean('is_completed')->default(false);
        $table->timestamps();
    });
}

Step 2: เขียน Controller (CRUD Logic)

class TaskController extends Controller {
    // แสดงรายการ
    public function index() {
        $tasks = Task::latest()->get();
        return view('tasks.index', compact('tasks'));
    }

    // บันทึก
    public function store(Request $request) {
        $validated = $request->validate(['title' => 'required|min:3']);
        Task::create($validated);
        return back()->with('success', 'บันทึกสำเร็จ');
    }

    // ลบ
    public function destroy(Task $task) {
        $task->delete();
        return back();
    }
}

Step 3: ออกแบบ View ด้วย Blade

<form action="{{ route('tasks.store') }}" method="POST">
    @csrf
    <input type="text" name="title">
    <button type="submit">เพิ่มงาน</button>
</form>

<ul>
    @foreach($tasks as $task)
        <li>
            {{ $task->title }}
            <form action="{{ route('tasks.destroy', $task) }}" method="POST">
                @csrf @method('DELETE')
                <button type="submit">ลบ</button>
            </form>
        </li>
    @endforeach
</ul>
