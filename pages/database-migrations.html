Database, Migrations & Seeding

การจัดการฐานข้อมูลใน Laravel ไม่ใช่เรื่องยากอีกต่อไป ด้วยระบบ Migration และ Seeding ที่ทรงพลัง

1. Migrations ขั้นสูง

นอกจากการสร้าง Table ใหม่แล้ว เรายังสามารถแก้ไข Table ที่มีอยู่ได้โดยไม่ต้องลบข้อมูลทิ้ง

การเพิ่ม Column ใหม่ลงใน Table ที่มีอยู่

// ใช้ artisan สร้าง migration ใหม่
php artisan make:migration add_phone_to_users_table --table=users

ในไฟล์ migration ที่ถูกสร้างขึ้น:

public function up()
{
    Schema::table('users', function (Blueprint $table) {
        $table->string('phone')->nullable()->after('email'); // เพิ่มหลัง email
    });
}

public function down()
{
    Schema::table('users', function (Blueprint $table) {
        $table->dropColumn('phone'); // เมื่อ Rollback ให้ลบออก
    });
}

2. Model Factories & Faker

Factories คือ "แม่พิมพ์" สำหรับสร้างข้อมูลทดสอบ โดยใช้ Library ชื่อว่า Faker

// database/factories/PostFactory.php

public function definition()
{
    return [
        'title' => $this->faker->sentence(),
        'content' => $this->faker->paragraphs(3, true),
        'user_id' => User::factory(), // สร้าง user ใหม่และผูกความสัมพันธ์อัตโนมัติ!
        'status' => $this->faker->randomElement(['draft', 'published']),
    ];
}

3. Database Seeding เจาะลึก

การรัน Seeder เพื่อเติมข้อมูลลงฐานข้อมูลจริงในตอนที่พัฒนา

เรียกใช้งาน Factory ใน Seeder

// database/seeders/DatabaseSeeder.php

public function run()
{
    // สร้าง User 5 คน และในแต่ละคนให้สร้าง Post 3 บทความ
    User::factory(5)
        ->has(Post::factory()->count(3))
        .create();
}

คำสั่งรัน Seeder

// รัน Seeder ทั้งหมด
php artisan db:seed

// ล้างฐานข้อมูลและ migrate ใหม่พร้อมรัน seed ทันที (นิยมใช้มากที่สุด)
php artisan migrate:fresh --seed
ข้อควรระวัง: คำสั่ง migrate:fresh จะลบข้อมูลใน Database ทั้งหมด! ห้ามใช้บน Production เด็ดขาด
