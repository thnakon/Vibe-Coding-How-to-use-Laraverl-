Collections & Helpers

ตัวช่วยที่จะทำให้ชีวิตการเขียน PHP ของคุณง่ายขึ้นอย่างมหาศาล

1. Laravel Collections

Collections คือ Wrapper สำหรับจัดการ Array ที่มาพร้อมกับ Method ช่วยเหลือมากมาย

$collection = collect(['alice', 'bob', 'charlie', null])
    ->filter() // กรองค่าว่างออก
    ->map(function ($name) {
        return strtoupper($name); // เปลี่ยนเป็นตัวใหญ่
    })
    ->sort();

// ผลลัพธ์: ['ALICE', 'BOB', 'CHARLIE']

2. String Helpers

การจัดการข้อความจะเป็นเรื่องสนุกด้วย Str class

use Illuminate\Support\Str;

$slug = Str::slug('Laravel 11 Tutorial For Beginners');
// ผลลัพธ์: laravel-11-tutorial-for-beginners

$limit = Str::limit('เนื้อหายาวมากๆ ที่เราต้องการตัดให้สั้นลง...', 20);
// ผลลัพธ์: เนื้อหายาวมากๆ ที่เรา...

3. Number Helpers

use Illuminate\Support\Number;

$currency = Number::currency(1500, in: 'THB', locale: 'th');
// ผลลัพธ์: 1,500.00 ฿
ข้อแนะนำ: อย่าเขียนโค้ด Array พื้นฐานด้วย PHP ธรรมดาจนเหนื่อย ลองเช็ค Laravel Documentation เรื่อง Collections ก่อนเสมอครับ!
